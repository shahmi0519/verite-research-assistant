# Send a greeting:

curl.exe -X POST http://localhost:8000/chat `
  -H "Content-Type: application/json" `
  -d '{\"message\": \"Hello!\", \"user_id\": \"test_user\"}'




# Ask a Verité question (copy the session_id)

  -H "Content-Type: application/json" `
  -d '{\"message\": \"What does Verité say about labour rights?\", \"user_id\": \"test_user\", \"session_id\": \"09e71bc5-7350-4f54-9c2e-d97330d97c73\"}'
